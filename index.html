# Group 6 – Shopping Cart System
# BE II Sem CSE-A | A.Y 2025-2026

cart = {}  # empty dictionary

# ── CONDITION 1: Add product and price ──
def add_product(name, price):
    cart[name] = price
    print(f"✅ '{name}' added at ₹{price:.2f}")

# ── CONDITION 2: Remove product ──
def remove_product(name):
    if name in cart:
        del cart[name]
        print(f"🗑️ '{name}' removed.")
    else:
        print(f"⚠️ '{name}' not found.")

# ── CONDITION 3: Calculate total bill ──
def calculate_total():
    if not cart:
        print("Cart is empty!"); return
    for p, v in cart.items():
        print(f"  {p:<20} ₹{v:.2f}")
    total = sum(cart.values())
    print(f"TOTAL BILL: ₹{total:.2f}")
Each function maps directly to one assignment condition. The entire cart state lives in a single Python dictionary.
